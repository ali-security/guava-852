048/**
049 * Represents an <a href="http://en.wikipedia.org/wiki/Internet_media_type">Internet Media Type</a>
050 * (also known as a MIME Type or Content Type). This class also supports the concept of media ranges
051 * <a href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.1">defined by HTTP/1.1</a>.
052 * As such, the {@code *} character is treated as a wildcard and is used to represent any acceptable
053 * type or subtype value. A media type may not have wildcard type with a declared subtype. The
054 * {@code *} character has no special meaning as part of a parameter. All values for type, subtype,
055 * parameter attributes or parameter values must be valid according to RFCs <a
056 * href="https://tools.ietf.org/html/rfc2045">2045</a> and <a
057 * href="https://tools.ietf.org/html/rfc2046">2046</a>.
058 *
059 * <p>All portions of the media type that are case-insensitive (type, subtype, parameter attributes)
060 * are normalized to lowercase. The value of the {@code charset} parameter is normalized to
061 * lowercase, but all others are left as-is.
062 *
063 * <p>Note that this specifically does <strong>not</strong> represent the value of the MIME {@code
064 * Content-Type} header and as such has no support for header-specific considerations such as line
065 * folding and comments.
066 *
067 * <p>For media types that take a charset the predefined constants default to UTF-8 and have a
068 * "_UTF_8" suffix. To get a version without a character set, use {@link #withoutParameters}.
069 *
070 * @since 12.0
071 * @author Gregory Kick
072 */
073@GwtCompatible
074@Immutable
075@ElementTypesAreNonnullByDefault
076public final class MediaType {
077  private static final String CHARSET_ATTRIBUTE = "charset";
078  private static final ImmutableListMultimap<String, String> UTF_8_CONSTANT_PARAMETERS =
079      ImmutableListMultimap.of(CHARSET_ATTRIBUTE, Ascii.toLowerCase(UTF_8.name()));

082  private static final CharMatcher TOKEN_MATCHER =
083      ascii()
084          .and(javaIsoControl().negate())
085          .and(CharMatcher.isNot(' '))
086          .and(CharMatcher.noneOf("()<>@,;:\\\"/[]?="));

106  private static final Map<MediaType, MediaType> KNOWN_TYPES = Maps.newHashMap();
107
108  private static MediaType createConstant(String type, String subtype) {
109    MediaType mediaType =
110        addKnownType(new MediaType(type, subtype, ImmutableListMultimap.<String, String>of()));
111    mediaType.parsedCharset = Optional.absent();
112    return mediaType;
113  }
114
115  private static MediaType createConstantUtf8(String type, String subtype) {
116    MediaType mediaType = addKnownType(new MediaType(type, subtype, UTF_8_CONSTANT_PARAMETERS));
117    mediaType.parsedCharset = Optional.of(UTF_8);
118    return mediaType;
119  }
120
121  private static MediaType addKnownType(MediaType mediaType) {
122    KNOWN_TYPES.put(mediaType, mediaType);
123    return mediaType;
124  }

838  /**
839   * Returns an optional charset for the value of the charset parameter if it is specified.
840   *
841   * @throws IllegalStateException if multiple charset values have been set for this media type
842   * @throws IllegalCharsetNameException if a charset value is present, but illegal
843   * @throws UnsupportedCharsetException if a charset value is present, but no support is available
844   *     in this instance of the Java virtual machine
845   */
846  public Optional<Charset> charset() {
847    // racy single-check idiom, this is safe because Optional is immutable.
848    Optional<Charset> local = parsedCharset;
849    if (local == null) {
850      String value = null;
851      local = Optional.absent();
852      for (String currentValue : parameters.get(CHARSET_ATTRIBUTE)) {
853        if (value == null) {
854          value = currentValue;
855          local = Optional.of(Charset.forName(value));
856        } else if (!value.equals(currentValue)) {
857          throw new IllegalStateException(
858              "Multiple charset values defined: " + value + ", " + currentValue);
859        }
860      }
861      parsedCharset = local;
862    }
863    return local;
864  }

883  /**
884   * <em>Replaces</em> all parameters with the given attribute with parameters using the given
885   * values. If there are no values, any existing parameters with the given attribute are removed.
886   *
887   * @throws IllegalArgumentException if either {@code attribute} or {@code values} is invalid
888   * @since 24.0
889   */
890  public MediaType withParameters(String attribute, Iterable<String> values) {
891    checkNotNull(attribute);
892    checkNotNull(values);
893    String normalizedAttribute = normalizeToken(attribute);
894    ImmutableListMultimap.Builder<String, String> builder = ImmutableListMultimap.builder();
895    for (Entry<String, String> entry : parameters.entries()) {
896      String key = entry.getKey();
897      if (!normalizedAttribute.equals(key)) {
898        builder.put(key, entry.getValue());
899      }
900    }
901    for (String value : values) {
902      builder.put(normalizedAttribute, normalizeParameterValue(normalizedAttribute, value));
903    }
904    MediaType mediaType = new MediaType(type, subtype, builder.build());
905    // if the attribute isn't charset, we can just inherit the current parsedCharset
906    if (!normalizedAttribute.equals(CHARSET_ATTRIBUTE)) {
907      mediaType.parsedCharset = this.parsedCharset;
908    }
909    // Return one of the constants if the media type is a known type.
910    return MoreObjects.firstNonNull(KNOWN_TYPES.get(mediaType), mediaType);
911  }

915   * given value. If multiple parameters with the same attributes are necessary use {@link
916   * #withParameters(String, Iterable)}. Prefer {@link #withCharset} for setting the {@code charset}
917   * parameter when using a {@link Charset} object.
918   *
919   * @throws IllegalArgumentException if either {@code attribute} or {@code value} is invalid
920   */
921  public MediaType withParameter(String attribute, String value) {

928   * charset} parameter will be present on the new instance regardless of the number set on this

947  /**
948   * Returns {@code true} if this instance falls within the range (as defined by <a
949   * href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html">the HTTP Accept header</a>) given
950   * by the argument according to three criteria:
951   *
952   * <ol>
953   *   <li>The type of the argument is the wildcard or equal to the type of this instance.
954   *   <li>The subtype of the argument is the wildcard or equal to the subtype of this instance.
955   *   <li>All of the parameters present in the argument are present in this instance.
956   * </ol>
957   *
958   * <p>For example:
959   *
960   * <pre>{@code
961   * PLAIN_TEXT_UTF_8.is(PLAIN_TEXT_UTF_8) // true
962   * PLAIN_TEXT_UTF_8.is(HTML_UTF_8) // false
963   * PLAIN_TEXT_UTF_8.is(ANY_TYPE) // true
964   * PLAIN_TEXT_UTF_8.is(ANY_TEXT_TYPE) // true
965   * PLAIN_TEXT_UTF_8.is(ANY_IMAGE_TYPE) // false
966   * PLAIN_TEXT_UTF_8.is(ANY_TEXT_TYPE.withCharset(UTF_8)) // true
967   * PLAIN_TEXT_UTF_8.withoutParameters().is(ANY_TEXT_TYPE.withCharset(UTF_8)) // false
968   * PLAIN_TEXT_UTF_8.is(ANY_TEXT_TYPE.withCharset(UTF_16)) // false
969   * }</pre>
970   *
971   * <p>Note that while it is possible to have the same parameter declared multiple times within a
972   * media type this method does not consider the number of occurrences of a parameter. For example,
973   * {@code "text/plain; charset=UTF-8"} satisfies {@code "text/plain; charset=UTF-8;
974   * charset=UTF-8"}.
975   */
976  public boolean is(MediaType mediaTypeRange) {
977    return (mediaTypeRange.type.equals(WILDCARD) || mediaTypeRange.type.equals(this.type))
978        && (mediaTypeRange.subtype.equals(WILDCARD) || mediaTypeRange.subtype.equals(this.subtype))
979        && this.parameters.entries().containsAll(mediaTypeRange.parameters.entries());
980  }

982  /**
983   * Creates a new media type with the given type and subtype.
984   *
985   * @throws IllegalArgumentException if type or subtype is invalid or if a wildcard is used for the
986   *     type, but not the subtype.
987   */
988  public static MediaType create(String type, String subtype) {
989    MediaType mediaType = create(type, subtype, ImmutableListMultimap.<String, String>of());
990    mediaType.parsedCharset = Optional.absent();
991    return mediaType;
992  }

994  private static MediaType create(
995      String type, String subtype, Multimap<String, String> parameters) {
996    checkNotNull(type);
997    checkNotNull(subtype);
998    checkNotNull(parameters);
999    String normalizedType = normalizeToken(type);
1000    String normalizedSubtype = normalizeToken(subtype);
1001    checkArgument(
1002        !WILDCARD.equals(normalizedType) || WILDCARD.equals(normalizedSubtype),
1003        "A wildcard type cannot be used with a non-wildcard subtype");
1004    ImmutableListMultimap.Builder<String, String> builder = ImmutableListMultimap.builder();
1005    for (Entry<String, String> entry : parameters.entries()) {
1006      String attribute = normalizeToken(entry.getKey());
1007      builder.put(attribute, normalizeParameterValue(attribute, entry.getValue()));
1008    }
1009    MediaType mediaType = new MediaType(normalizedType, normalizedSubtype, builder.build());
1010    // Return one of the constants if the media type is a known type.
1011    return MoreObjects.firstNonNull(KNOWN_TYPES.get(mediaType), mediaType);
1012  }

1068  private static String normalizeToken(String token) {
1069    checkArgument(TOKEN_MATCHER.matchesAllOf(token));
1070    checkArgument(!token.isEmpty());
1071    return Ascii.toLowerCase(token);
1072  }

1074  private static String normalizeParameterValue(String attribute, String value) {
1075    checkNotNull(value); // for GWT
1076    checkArgument(ascii().matchesAllOf(value), "parameter values must be ASCII: %s", value);
1077    return CHARSET_ATTRIBUTE.equals(attribute) ? Ascii.toLowerCase(value) : value;
1078  }

1080  /**
1081   * Parses a media type from its string representation.
1082   *
1083   * @throws IllegalArgumentException if the input is not parsable
1084   */
1085  @CanIgnoreReturnValue // TODO(b/219820829): consider removing
1086  public static MediaType parse(String input) {
1087    checkNotNull(input);
1088    Tokenizer tokenizer = new Tokenizer(input);
1089    try {
1090      String type = tokenizer.consumeToken(TOKEN_MATCHER);
1091      consumeSeparator(tokenizer, '/');
1092      String subtype = tokenizer.consumeToken(TOKEN_MATCHER);
1093      ImmutableListMultimap.Builder<String, String> parameters = ImmutableListMultimap.builder();
1094      while (tokenizer.hasMore()) {
1095        consumeSeparator(tokenizer, ';');
1096        String attribute = tokenizer.consumeToken(TOKEN_MATCHER);
1097        consumeSeparator(tokenizer, '=');
1098        String value;
1099        if ('"' == tokenizer.previewChar()) {
1100          tokenizer.consumeCharacter('"');
1101          StringBuilder valueBuilder = new StringBuilder();
1102          while ('"' != tokenizer.previewChar()) {
1103            if ('\\' == tokenizer.previewChar()) {
1104              tokenizer.consumeCharacter('\\');
1105              valueBuilder.append(tokenizer.consumeCharacter(ascii()));
1106            } else {
1107              valueBuilder.append(tokenizer.consumeToken(QUOTED_TEXT_MATCHER));
1108            }
1109          }
1110          value = valueBuilder.toString();
1111          tokenizer.consumeCharacter('"');
1112        } else {
1113          value = tokenizer.consumeToken(TOKEN_MATCHER);
1114        }
1115        parameters.put(attribute, value);
1116      }
1117      return create(type, subtype, parameters.build());
1118    } catch (IllegalStateException e) {
1119      throw new IllegalArgumentException("Could not parse '" + input + "'", e);
1120    }
1121  }

1123  private static void consumeSeparator(Tokenizer tokenizer, char c) {
1124    tokenizer.consumeTokenIfPresent(LINEAR_WHITE_SPACE);
1125    tokenizer.consumeCharacter(c);
1126    tokenizer.consumeTokenIfPresent(LINEAR_WHITE_SPACE);
1127  }

1129  private static final class Tokenizer {
1130    final String input;
1131    int position = 0;
1132
1133    Tokenizer(String input) {
1134      this.input = input;
1135    }
1136
1137    @CanIgnoreReturnValue
1138    String consumeTokenIfPresent(CharMatcher matcher) {
1139      checkState(hasMore());
1140      int startPosition = position;
1141      position = matcher.negate().indexIn(input, startPosition);
1142      return hasMore() ? input.substring(startPosition, position) : input.substring(startPosition);
1143    }
1144
1145    String consumeToken(CharMatcher matcher) {
1146      int startPosition = position;
1147      String token = consumeTokenIfPresent(matcher);
1148      checkState(position != startPosition);
1149      return token;
1150    }
1151
1152    char consumeCharacter(CharMatcher matcher) {
1153      checkState(hasMore());
1154      char c = previewChar();
1155      checkState(matcher.matches(c));
1156      position++;
1157      return c;
1158    }
1159
1160    @CanIgnoreReturnValue
1161    char consumeCharacter(char c) {
1162      checkState(hasMore());
1163      checkState(previewChar() == c);
1164      position++;
1165      return c;
1166    }
1167
1168    char previewChar() {
1169      checkState(hasMore());
1170      return input.charAt(position);
1171    }
1172
1173    boolean hasMore() {
1174      return (position >= 0) && (position < input.length());
1175    }
1176  }

1204  private static final MapJoiner PARAMETER_JOINER = Joiner.on("; ").withKeyValueSeparator("=");
1205
1206  /**
1207   * Returns the string representation of this media type in the format described in <a
1208   * href="http://www.ietf.org/rfc/rfc2045.txt">RFC 2045</a>.
1209   */
1210  @Override
1211  public String toString() {
1212    // racy single-check idiom, safe because String is immutable
1213    String result = toString;
1214    if (result == null) {
1215      result = computeToString();
1216      toString = result;
1217    }
1218    return result;
1219  }
1220
1221  private String computeToString() {
1222    StringBuilder builder = new StringBuilder().append(type).append('/').append(subtype);
1223    if (!parameters.isEmpty()) {
1224      builder.append("; ");
1225      Multimap<String, String> quotedParameters =
1226          Multimaps.transformValues(
1227              parameters,
1228              (String value) ->
1229                  (TOKEN_MATCHER.matchesAllOf(value) && !value.isEmpty())
1230                      ? value
1231                      : escapeAndQuote(value));
1232      PARAMETER_JOINER.appendTo(builder, quotedParameters.entries());
1233    }
1234    return builder.toString();
1235  }
1236
1237  private static String escapeAndQuote(String value) {
1238    StringBuilder escaped = new StringBuilder(value.length() + 16).append('"');
1239    for (int i = 0; i < value.length(); i++) {
1240      char ch = value.charAt(i);
1241      if (ch == '\r' || ch == '\\' || ch == '"') {
1242        escaped.append('\\');
1243      }
1244      escaped.append(ch);
1245    }
1246    return escaped.append('"').toString();
1247  }
